po_depose_pretender= {
	# Show in treaties tab
	category = 6
	
	prestige_base = 0.8
	ae_base = 0.0
	
	power_projection = humiliated_rival
	power_cost_base = 1.0
	
	# Warscore cost scales directly according to the warscore cost of the provinces that the owner retains
	# IE: If you take some provinces, the cost of forced conversion is reduced
	warscore_cost = {
		no_provinces = 100
		owner_keeps = 0.0 # The provinces that are not of any of the types below
		conquered = 0.0 # Provinces that are selected by a Demand Province treaty
		returned_core = 0.0 # Provinces that are selected by a Return Core treaty
		released_nation = 0.0 # Provinces that will become a part of a selected released-to-be nation
		cancelled_subject = 0.0 # This one should not be relevant here since Giver can't be a cancelled subject. Including anyway for completeness
		concede_colonial_region = 0.0 # Ditto

	}
	
	# No limit to the cost of forced conversion (ie. large nations cannot be forcibly converted)
	warscore_cap = 100
	
	requires_is_allowed = no
	
    is_visible = {
		is_in_war = {
			attacker_leader = ROOT
			defender_leader = emperor
			casus_belli = cb_depose_the_pretender
		}
		FROM = {
			is_emperor = yes
		}
    }
	
	is_allowed = {
		always = yes
	}
	
	effect = {
		FROM = {
			add_imperial_influence = -100
			add_stability = -1
			add_legitimacy = -50
			add_country_modifier = {
				name = wallachia_forcibly_deposed_by_outside_power
				duration = 9125
			}
			set_country_flag = ignore_for_select_new_emperor_effect
		}
		depose_the_pretender_effect = yes
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		if = {
			limit = {
				is_rival = FROM
			}
			add_humiliation_reward = yes
		}
		set_country_flag = deposed_hre_emperor
		save_global_event_target_as = dude_who_deposed_emperor
		country_event = {
			id = wallachia_hre_events.0
			days = 1
		}
	}
	
	ai_weight = {
		export_to_variable = {
			variable_name = ai_value
			value = 50
		}
	}
	
}

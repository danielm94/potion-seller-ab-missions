estate_nobles_integrate_german_nobility = {
	icon = privilege_grant_autonomy
	max_absolutism = -15	#Change in max absolutism while this privilege is enacted
	loyalty = -0.05
	influence = 0.1
	is_valid = {
		has_country_flag = unlock_integrating_german_nobility_privilege
	}
	can_select = {
		any_owned_province = {
			OR = {
				culture_group = germanic
				culture_group = united_germanic
			}
		}
	}
	on_granted = {
		custom_tooltip = every_owned_germanic_province_tooltip
		tooltip = {
			add_province_modifier = {
				name = tuh_integrated_germanic_nobility
				duration = -1
			}
			custom_tooltip = integrate_german_nobility_crownland_tooltip
			custom_tooltip = AB_EMPTY
			custom_tooltip = this_will_apply_to_all_german_provinces_conquered_tooltip
		}
	}
	on_granted_province = {
		if = {
			limit = {
				OR = {
					culture_group = germanic
					culture_group = united_germanic
				}
			}
			add_province_modifier = {
				name = tuh_integrated_germanic_nobility
				duration = -1
			}
			hidden_effect = {
				export_to_variable = {
					which = province_dev_var
					value = development
				}
				multiply_variable = {
					which = province_dev_var
					value = 0.05
				}
				owner = {
					set_variable = {
						which = province_dev_var
						which = PREV
					}
					variable_change_estate_land_share = {
						estate = estate_nobles
						var = province_dev_var
					}
				}
			}
		}
	}
	on_revoked_province = {
		if = {
			limit = {
				has_province_modifier = tuh_integrated_germanic_nobility
			}
			remove_province_modifier = tuh_integrated_germanic_nobility
		}
	}
	conditional_modifier = {
		trigger = {
			has_government_attribute = no_absolutism_from_boier_privileges
		}
		modifier = {
			max_absolutism = 15
		}
	}
	ai_will_do = {
		factor = 10
	}
}

estate_burghers_wal_carpathian_expansion = {
	icon = privilege_dev_cost
	max_absolutism = -10
	influence = 0.1
	loyalty = 0.05
	land_share = 5
	is_valid = {
		has_unlocked_estate_privilege = {
			estate_privilege = estate_burghers_wal_carpathian_expansion
		}
	}
	can_select = {
		any_owned_province = {
			region = carpathia_region
			OR = {
				has_terrain = highlands
				has_terrain = mountain
				has_terrain = hills
			}
		}
	}
	on_granted = {
		custom_tooltip = carpathian_expansion_on_granted_tooltip
	}
	on_granted_province = {
		if = {
			limit = {
				has_terrain = mountain
				region = carpathia_region
			}
			add_province_modifier = {
				name = wallachia_carpathian_mountain_expansion
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_terrain = highlands
				region = carpathia_region
			}
			add_province_modifier = {
				name = wallachia_carpathian_highlands_expansion
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_terrain = hills
				region = carpathia_region
			}
			add_province_modifier = {
				name = wallachia_carpathian_hills_expansion
				duration = -1
			}
		}
	}
	on_revoked = {
		custom_tooltip = carpathian_expansion_on_revoked_tooltip
	}
	on_revoked_province = {
		if = {
			limit = {
				OR = {
					has_province_modifier = wallachia_carpathian_mountain_expansion
					has_province_modifier = wallachia_carpathian_highlands_expansion
					has_province_modifier = wallachia_carpathian_hills_expansion
				}
			}
			remove_province_modifier = wallachia_carpathian_mountain_expansion
			remove_province_modifier = wallachia_carpathian_highlands_expansion
			remove_province_modifier = wallachia_carpathian_hills_expansion
		}
	}
	on_invalid = {
	}
	on_invalid_province = {
		if = {
			limit = {
				OR = {
					has_province_modifier = wallachia_carpathian_mountain_expansion
					has_province_modifier = wallachia_carpathian_highlands_expansion
					has_province_modifier = wallachia_carpathian_hills_expansion
				}
			}
			remove_province_modifier = wallachia_carpathian_mountain_expansion
			remove_province_modifier = wallachia_carpathian_highlands_expansion
			remove_province_modifier = wallachia_carpathian_hills_expansion
		}
	}
	penalties = {
	}
	benefits = {
	}
	ai_will_do = {
		factor = 100
		modifier = {
			factor = 2
			num_of_owned_provinces_with = {
				OR = {
					has_terrain = mountain
					has_terrain = highlands
					has_terrain = hills
				}
				region = carpathia_region
				value = 8
			}
		}
		modifier = {
			factor = 0
			is_absolutism_active = yes
		}
	}
}

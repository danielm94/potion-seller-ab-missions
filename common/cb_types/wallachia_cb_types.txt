cb_eastern_roman_ambition = {
	valid_for_subject = no
	prerequisites_self = {
		has_country_flag = eastern_roman_ambition
	}
	prerequisites = {
		OR = {
			is_neighbor_of = FROM
			AND = {
				FROM = {
					any_owned_province = {
						province_id = 321
					}
				}
				is_neighbor_of_province = 1317
			}
			AND = {
				FROM = {
					any_owned_province = {
						area = aegean_archipelago_area
					}
				}
				is_neighbor_of_province = 1319
			}
		}
		FROM = {
			any_owned_province = {
				eastern_roman_empire_province_cb_trigger = yes
			}
		}
	}
	war_goal = eastern_roman_wargoal
}

cb_depose_the_pretender = {
	valid_for_subject = no
	prerequisites_self = {
		government_rank = 3
		OR = {
			owns_core_province = 151
			owns_core_province = 118
		}
		has_country_flag = depose_hre_emperor_cb
		OR = {
			NOT = {
				has_country_flag = deposed_hre_emperor
			}
			had_country_flag = {
				flag = deposed_hre_emperor
				days = 5475
			}
		}
	}
	attacker_disabled_po = {
		po_demand_provinces
		po_release_vassals
		po_release_annexed
		po_return_cores
		po_become_vassal
		po_subjugate_vassal
		po_become_tributary_state
		po_subjugate_tributary_state
		po_form_personal_union
		po_transfer_vassals
	}
	prerequisites = {
		FROM = {
			is_emperor = yes
		}
	}
	war_goal = depose_the_pretender_wargoal
}

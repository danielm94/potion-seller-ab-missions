	id = wallachian_civil_war_events.16
	title = wallachian_civil_war_events.16.t
	desc = {
		desc = wallachian_civil_war_events_refused.16.d
		trigger = {
			has_refused_boiermiea_ultimatum = yes
		}
	}
	desc = {
		desc = wallachian_civil_war_events_butchered.16.d
		trigger = {
			has_butchered_the_boiermiea = yes
		}
	}
	immediate = {
		if = {
			limit = {
				has_refused_boiermiea_ultimatum = yes
			}
			hidden_effect = {
				random_owned_province = {
					spawn_large_scaled_rebels = {
						type = wallachia_civil_war_danesti_refused_ultimatum_rebels
						leader = "Mircea"
						leader_dynasty = "Danesti"
					}
				}
				set_country_flag = wallachian_civil_war_danesti_rebels_spawned
			}
		}
		else = {
			hidden_effect = {
				random_owned_province = {
					spawn_large_scaled_rebels = {
						type = wallachia_civil_war_danesti_butchered_rebels
						leader = "Barbu"
						leader_dynasty = "Danesti"
					}
				}
				set_country_flag = wallachian_civil_war_danesti_rebels_spawned
			}
		}
	}
	picture = AB_LOTHAIR_eventPicture
	is_triggered_only = yes
	option = {
		name = wallachian_civil_war_events.16_option
	}
}

#Racovita Bribe Event (Your example, for reference - Event ID .19)
country_event = {
	id = wallachian_civil_war_events.19
	title = wallachian_civil_war_events.19.t
	desc = wallachian_civil_war_events.19.d
	picture = AB_LOTHAIR_eventPicture
	is_triggered_only = yes
	option = {
		name = wallachian_civil_war_give_bribe_for_loyalty		# Generic option name
		add_years_of_income = -0.5
		add_legitimacy = 5
		custom_tooltip = wallachian_civil_war_racovita_loyalty_tooltip
		hidden_effect = {
			set_country_flag = wallachian_civil_war_defeated_racovita
		}
	}
	option = {
		name = wallachian_civil_war_refuse_loyalty		# Generic option name
		add_manpower = -0.001		# Small negative impact to represent displeasure/consequence
		add_legitimacy = -5
		add_prestige = -5
	}
}

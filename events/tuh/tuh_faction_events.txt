country_event = {
	id = tuh_faction_events.2
	title = tuh_faction_events.2.t
	desc = tuh_faction_events.2.d
	picture = BYZ_BYZANTINE_EAGLE_eventPicture
	is_triggered_only = no
	fire_only_once = yes
	mean_time_to_happen = {
		months = 36
		modifier = {
			factor = 0.5
			faction_in_power = cantacuzino_faction
		}
		modifier = {
			factor = 0.5
			BYZ = {
				NOT = {
					legitimacy_equivalent = 50
				}
			}
		}
	}
	trigger = {
		NOT = {
			has_country_flag = tuh_claimed_a_byzantine_throne
		}
		has_government_attribute = romanian_elective_monarchy
		exists = BYZ
		dynasty = "Cantacuzino"
		marriage_with = BYZ
		BYZ = {
			has_heir = no
		}
		has_more_legitimacy_or_equivalent_than = {
			targetA = ROOT
			targetB = BYZ
		}
		has_more_prestige_than = {
			targetA = ROOT
			targetB = BYZ
		}
		army_strength = {
			who = BYZ
			value = 1.1
		}
	}
	option = {
		name = tuh_faction_events.2.a
		add_casus_belli = {
			type = cb_restore_personal_union
			target = BYZ
			months = 120
		}
		add_faction_influence = {
			faction = cantacuzino_faction
			influence = 20
		}
		BYZ = {
			add_opinion = {
				who = PREV
				modifier = claimed_throne_on_us
			}
			break_alliance = PREV
		}
		every_country = {
			limit = {
				marriage_with = PREV
				NOT = {
					tag = BYZ
				}
			}
			add_opinion = {
				who = PREV
				modifier = claimed_throne_other_rm
			}
		}
		set_country_flag = tuh_claimed_a_byzantine_throne
	}
	option = {
		name = tuh_faction_events.2.b
		add_prestige = -5
	}
}
